##  insert 함수 사용하여 원하는 위치에 추가하기

- insert(location, column, value, allow_duplicates)


```python

# column명은 Fare10
# value는 기존 Fare라는 컬럼을 10으로 나눈 값
# allow_duplicates= False가 default

train_data.insert(3, 'Fare10', train_data['Fare'] / 10)
train_data.head()
```
